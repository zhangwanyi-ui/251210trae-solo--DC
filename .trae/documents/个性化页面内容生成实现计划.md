### ContentArea组件修改
```javascript
// src/components/ContentArea.jsx
import pageContentConfig from '../config/pageContentConfig';

const ContentArea = ({ title, navType }) => {
  // 解析页面标题，获取左侧导航页名称和内容导航项
  const parseTitle = (title) => {
    if (title.includes(' - ')) {
      const [mainTitle, subTitle] = title.split(' - ');
      return { mainTitle, subTitle };
    }
    return { mainTitle: title, subTitle: '' };
  };

  const { mainTitle, subTitle } = parseTitle(title);
  const contentConfig = pageContentConfig[mainTitle] || {};
  const typeConfig = contentConfig[navType] || {};

  // 根据配置生成内容
  const renderContent = () => {
    switch (navType) {
      case 'monitor':
        return renderMonitorContent(typeConfig, mainTitle, subTitle);
      case 'report':
        return renderReportContent(typeConfig, mainTitle, subTitle);
      case 'list':
        return renderListContent(typeConfig, mainTitle, subTitle);
      case 'form':
        return renderFormContent(typeConfig, mainTitle, subTitle);
      default:
        return renderDefaultContent(title);
    }
  };

  // ... 各类型内容渲染函数
};
```

### 内容模板实现
- **监控类页面**：根据配置的指标名称生成指标卡片，根据图表类型生成对应的图表
- **报告类页面**：根据配置的报告章节生成报告内容，包含图表和数据分析
- **列表类页面**：根据配置的列名生成表格，根据数据源生成示例数据
- **表单类页面**：根据配置的字段生成表单，包含输入框、下拉选择等控件
